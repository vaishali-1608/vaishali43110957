#inner join
SELECT CSE.regno, CSE.name AS CSE_Name, ECE.name AS ECE_Name FROM CSE INNER JOIN ECE ON CSE.regno = ECE.regno;

#left join
SELECT CSE.regno, CSE.name AS CSE_Name, ECE.name AS ECE_Name FROM CSE LEFT JOIN ECE ON CSE.regno = ECE.regno;

#right join
SELECT CSE.regno, CSE.name AS CSE_Name, ECE.name AS ECE_Name FROM CSE RIGHT JOIN ECE ON CSE.regno = ECE.regno;

#full join
SELECT CSE.regno, CSE.name AS CSE_Name, ECE.name AS ECE_Name FROM CSE FULL JOIN ECE ON CSE.regno = ECE.regno;

#cross join
SELECT CSE.name AS CSE_Name, ECE.name AS ECE_Name FROM CSE CROSS JOIN ECE;

#self join
SELECT A.name AS Student1, B.name AS Student2 FROM CSE A, CSE B WHERE A.regno < B.regno AND A.dept = B.dept;
